009    package jcolibri.exception;
010    
011    /**
012     * Trying to apply a similarity function to an incompatible attribute
013     * 
014     * @author Juan A. Recio-García
015     * @version 2.0
016     */
017    public class NoApplicableSimilarityFunctionException extends Exception {
018            private static final long serialVersionUID = 1L;
019    
020            public NoApplicableSimilarityFunctionException(Class sf, Class c)
021            {
022                    super("SimilarityFunction: "+sf+" not applicable to type: "+c);
023            }
024            
025            public NoApplicableSimilarityFunctionException(String msg)
026            {
027                    super(msg);
028            }
029            
030            public NoApplicableSimilarityFunctionException(Exception ex)
031            {
032                    super(ex);
033            }
034    }
